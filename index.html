$(document).ready(function(){
	$("selector").VCarouselMOD({
		playBack: true,
		step: 2,
		ID: 3,
		play: 3000,
		speed: 1000,
		navHidden: true,
		navNum: true,
		fading: false,
		easing: "easeOutBounce",
		visible: 3,
		lightbox: true,
		LBImgN: true,
		LBcaption: ".carousel-caption",
		LBdesc: ".carousel-desc",
		LBaddLink: ".carousel-linkForLB",
		prevText: "< Prev",
		nextText: "Next >",
		LBprevText: "Prev",
		LBnextText: "Next",
		control: ".carousel-control",
		LBloadImg: ".carousel-link",
		navigation: ".carousel-nav",
		navigationImg: true
	});
});
